%\renewenvironment{proof}{\par\textbf{证明.}\;}{\qed\par}

% 缺点：\bf 是过时命令，可以用 textb f等替代，但编译会有关于字体的警告，不过不影响使用【技术有限，暂时无法解决】

\newenvironment{solution}{\begin{proof}[\indent\bf 解]}{\end{proof}}
